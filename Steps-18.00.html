DAX Calculation 18  For OFFER RATE / INTERVIEW RATE / PRE_YEAR_INTERVIEW_RATE / PRE_YEAR_OFFER_RATE

       We need to format the data to percentage

    - (1) Click on the measure that we need to format to a percentage

    - (2) Click Modeling

    - (3) Click “%” sign

    

    Measures we need to create

     Days: Time to fill. Average time since the job open to close

     DAYS = var day =  AVERAGE(Fact_JobReferral[DURATION])

      return IF(day > 0,day,0)

     TOTAL JOB: Calculate the number of JOB posted.

     TOTAL JOB = var job = CALCULATE(DISTINCTCOUNT(Dim_JRP_Jobs[ID]),

                                                FILTER(Dim_JRP_Jobs,Dim_JRP_Jobs[ID] <> BLANK())

                                                Return if ( job > 0, job, 0)


    TOTAL SUBMISSION: Calculate the number of Candidate’ SUBMISSIONS.

     TOTAL SUBMISSION = DISTINCTCOUNT('Fact_Submission_Interview_Offer'[SMI_SubID])


    TOTAL INTERVIEW: Calculate the number of INTERVIEWS that take by Candidates.

     TOTAL INTERVIEW = CALCULATE(DISTINCTCOUNT('Fact_Submission_Interview_Offer'[FormInterviewIer',

                                                           Fact_Submission_Interview_Offer',

                                                            'Fact-Submission_offer'[formInterviewID]<>BLANK())))

         TOTAL OFFER: Calculate the number of OFFERS for Candidates.

     TOTAL OFFER   = CALCULATE(DISTINCTCOUNT('Fact_Submission_Interview_Offer'[OfferID]),

                                                       FILTER('Fact_Submission_Interview_Offer',

                                                      'Fact_S     ubmission_Interview_Offer'[OfferID] <> BLANK()))

    INTERVIEW RATE:  Percentage of INTERVIEW based on SUBMISSIONS.

      INTERVIEW RATE =

      var INTERVIEW_RATE = DIVIDE([TOTAL INTERVIEW],[TOTAL SUBMISSION],0)

      return IF(INTERVIEW_RATE > 0, INTERVIEW_RATE,0)


    AVG_INTERVIEW: Average INTERVIEWS that Candidates take to get OFFER.

       AVG_INTERVIEW =

       VAR AVG_INTERVIEW = DIVIDE([TOTAL INTERVIEW],[TOTAL OFFER])

       RETURN IF ( AVG_INTERVIEW > 0, AVG_INTERVIEW, 0 )


    OFFER RATE: Percentage of OFFER based on INTERVIEW

        OFFER RATE =

       var OFFER_RATE = DIVIDE([TOTAL OFFER],[TOTAL INTERVIEW],0)

      RETURN IF(OFFER_RATE > 0, OFFER_RATE,0)


    PRE_YEAR_INTERVIEW_RATE: Set as Target for INTERVIEW RATE 

       PRE_YEAR_INTERVIEW_RATE= CALCULATE([INTERVIEW RATE],PREVIOUSYEAR(Dim_JRP_Jobs[Year]))


    PRE_YEAR_AVG_INTERVIEW: Set as Target for AVG_INTERVIEW 

         PRE_YEAR_AVG_INTERVIEW = CALCULATE([AVG_INTERVIEW],PREVIOUSYEAR(Dim_JRP_Jobs [Year]))


    PRE_YEAR_OFFER_RATE: Set as Target for OFFER RATE

       PRE_YEAR_OFFER_RATE = CALCULATE([OFFER RATE],PREVIOUSYEAR(Dim_JRP_Jobs[Year]))
